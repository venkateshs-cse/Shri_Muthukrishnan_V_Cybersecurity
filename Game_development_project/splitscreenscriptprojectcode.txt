using UnityEngine;

public class SplitScreenPlayerController : MonoBehaviour
{
    [Header("Car Settings")]
    public float speed=15.0f;
    public float turnSpeed=25.0f;

    [Header("Controller Settings")]
     public KeyCode forwardKey;
     public KeyCode backwardKey;
     public KeyCode rightKey;
     public KeyCode leftKey;
    private float horizontalInput;
    private float verticalInput;
    // Start is called once before the first execution of Update after the MonoBehaviour is created
    void Start()
    {
        
    }

    // Update is called once per frame
    void Update()
    {    //Reset horizontal and vertical every frame
        horizontalInput=0.0f;
        verticalInput=0.0f;
        
       //getting input for forward and backward keys
        if(Input.GetKey(forwardKey)) verticalInput=1f;
        else if(Input.GetKey(backwardKey)) verticalInput=-1f;

       //getting input for left and rightkeys
        if(Input.GetKey(leftKey)) horizontalInput=-1f;
        else if(Input.GetKey(rightKey))horizontalInput=1f;

        //forward speed
        transform.Translate(Vector3.forward*speed*Time.deltaTime*verticalInput);

       //turning speed
        transform.Rotate(Vector3.up,turnSpeed*Time.deltaTime*horizontalInput);


    }
}
